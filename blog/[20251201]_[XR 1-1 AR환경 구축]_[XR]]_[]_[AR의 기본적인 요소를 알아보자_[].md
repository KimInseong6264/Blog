```
[SerializeField] GameObject bulletPrefab;
[SerializeField] ARRaycastManager raycastMgr;

public void shoot()
{
    Ray ray = new Ray();
    ray.origin = Camera.main.transform.position;
    ray.direction = Camera.main.transform.forward;

    List<ARRaycastHit> hits = new List<ARRaycastHit>();

    // AR의 레이캐스트는 2개의 인자를 받음, 레이하고, 그리고 충돌된 수많은 정보를 담을 리스트
    if (raycastMgr.Raycast(ray, hits))
    {
        Instantiate(bulletPrefab, hits[0].pose.position, hits[0].pose.rotation);
        // 충돌 여러개 인식하면 리스트 중, 0번째, 즉 처음 인식된 곳에 총알 생성
    }

}
```
